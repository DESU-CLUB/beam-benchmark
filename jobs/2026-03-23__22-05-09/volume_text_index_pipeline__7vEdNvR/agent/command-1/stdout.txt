```python
import modal
import os

app = modal.App("text-index-pipeline")

volume = modal.Volume.from_name("text-index-volume", environment_name="modal-vsdatagen")

@app.function(volumes={"/data": volume})
def index_documents(documents: list[str]):
    # Ensure the target directory exists
    os.makedirs("/data/index", exist_ok=True)
    
    # Write each document to a file
    for i, doc in enumerate(documents):
        with open(f"/data/index/doc_{i}.txt", "w") as f:
            f.write(doc)
    
    # Commit the volume to persist changes
    volume.commit()
    
    return len(documents)

@app.local_entrypoint()
def main():
    # Sample documents to index
    sample_docs = [
        "This is the first sample document for indexing.",
        "A second document that contains some text content.",
        "The third and final document in our initial test set."
    ]
    
    # Call the remote indexing function
    count = index_documents.remote(sample_docs)
    
    # Store the count in a Modal Dict
    modal.Dict.from_name("text-index-output", create_if_missing=True).put("doc_count", count)
    
    # Print the count
    print(f"Indexed {count} documents")
```
